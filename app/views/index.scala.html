function getPages() {
    var url = 'https://www.google.com/search?client=ubuntu&channel=fs&q=nashorn+scala+class&ie=utf-8&oe=utf-8#channel=fs&q=nashorn+create+java+object';
    var document = new Document(url);
    var result = [];
    var arr = document.attr('a.fl','href');
    for(var i in arr) {
        result.push( document.absolute(arr[i], 'https://www.google.com/') )
    }
    return result;
}

function scrap(url, document) {
    var result = [];
    var arr = document.attr('img','src');
    for(var i in arr) {
        result.push( document.absolute(arr[i], url) )
    }
    return result;
}
